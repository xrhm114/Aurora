```matlab
t = 0:0.001:3*2*pi;    % 时间向量
f = 1;                 % 频率
y = cos(2*f*t).^2;     % 函数
figure;
hold on;
area(t, y, 'FaceColor', 'w'); % 首先用白色填充整个图形下方区域
area(t(t<=2*pi), y(t<=2*pi), 'FaceColor', 'y'); % 然后用黄色填充0到2π的积分面积
plot(t, y, 'b');       % 绘制余弦平方函数
hold off;
```

t = 0:0.001:3*2*pi;    % 时间向量
f = 1;                 % 频率

hold on;
